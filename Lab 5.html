  group_by(STATE) %>%
  summarise(avg_temp = mean(temp, na.rm = TRUE))
avg_temp_by_state <- avg_temp_by_state %>%
  mutate(temp_level = case_when(
    avg_temp < 20 ~ "low",
    avg_temp >= 20 & avg_temp < 25 ~ "mid",
    avg_temp >= 25 ~ "high"
  ))
dat_with_levels <- merge(dat, avg_temp_by_state, by = "STATE")
